import matplotlib.pyplot as plt
import numpy as np
a=np.arange(1,8)
plt.xlim(-10,60)
plt.ylim(-0.5,8)
plt.plot(a**2,a,color="red",linewidth=10,label="squares")
plt.plot(a**3,a,color="green",linewidth=5,label="cubes")
plt.xlabel("X area")
plt.ylabel("Y area")
plt.title("TITLE")
plt.minorticks_on()
plt.grid(b="True",which="minor")
plt.legend()
plt.show()
